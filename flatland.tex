% constituent track components    
\newcommand{\trackblank}[2]{
    \draw[draw=darkgray!75, line width=0.15mm] (#1 - 0.5, #2 - 0.5) rectangle ++(1,1);
    \node[align=left] at (#1 +0.35,#2 +0.35) {\footnotesize{0}}; 
}

\newcommand{\trackhorizontal}[3][false]{
    \path[sleepers] (#2 - 0.5 , #3) to (#2 + 0.5 , #3);
    \path[rails] (#2 - 0.5 , #3) to (#2 + 0.5 , #3);
    \ifthenelse{\equal{#1}{border}}{
        \draw[draw=darkgray!75, line width=0.15mm, label={[blue]30:\small $1$}] (#2 - 0.5, #3 - 0.5) rectangle ++(1,1);
        \node[align=left] at (#2 +0.35,#3 +0.35) {\footnotesize{1}}; 
    }
}

% full track type
\newcommand{\track}[4]{
    % blank tracks
        \ifnum#1=0      \trackblank{#2}{#3}
        \fi

    % straight tracks
         \ifnum#1=32800 \trackvertical[#4]{#2}{#3}
    \else\ifnum#1=1025  \trackhorizontal[#4]{#2}{#3}
    \else               \relax
    \fi\fi

    % curved tracks
         \ifnum#1=4608  \trackcurvesw[#4]{#2}{#3}
    \else\ifnum#1=16386 \trackcurvese[#4]{#2}{#3}
    \else\ifnum#1=72    \trackcurvene[#4]{#2}{#3}
    \else\ifnum#1=2064  \trackcurvenw[#4]{#2}{#3}
    \else               \relax
    \fi\fi\fi\fi

    % simple switches
         \ifnum#1=37408 \vsw[#4]{#2}{#3}
    \else\ifnum#1=17411 \hse[#4]{#2}{#3}
    \else\ifnum#1=32872 \vne[#4]{#2}{#3}
    \else\ifnum#1=3089  \hnw[#4]{#2}{#3}
    \else               \relax
    \fi\fi\fi\fi

         \ifnum#1=5633  \hsw[#4]{#2}{#3}
    \else\ifnum#1=49186 \vse[#4]{#2}{#3}
    \else\ifnum#1=1097  \hne[#4]{#2}{#3}
    \else\ifnum#1=34864 \vnw[#4]{#2}{#3}
    \else               \relax
    \fi\fi\fi\fi

    % crossing
         \ifnum#1=33825 \crossing[#4]{#2}{#3}
    \else               \relax
    \fi

    % slip switches
         \ifnum#1=38433 \hvsw[#4]{#2}{#3}
    \else\ifnum#1=50211 \hvse[#4]{#2}{#3}
    \else\ifnum#1=33897 \hvne[#4]{#2}{#3}
    \else\ifnum#1=35889 \hvnw[#4]{#2}{#3}
    \else               \relax
    \fi\fi\fi\fi

    % double-slip switches
         \ifnum#1=38505 \hvnesw[#4]{#2}{#3}
    \else\ifnum#1=52275 \hvnwse[#4]{#2}{#3}
    \else               \relax
    \fi\fi

    % symmetrical switches
         \ifnum#1=20994 \swse[#4]{#2}{#3}
    \else\ifnum#1=16458 \sene[#4]{#2}{#3}
    \else\ifnum#1=2136  \nenw[#4]{#2}{#3}
    \else\ifnum#1=6672  \nwsw[#4]{#2}{#3}
    \else               \relax
    \fi\fi\fi\fi
    

}
